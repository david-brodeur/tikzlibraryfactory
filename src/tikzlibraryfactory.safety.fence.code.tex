% \brief Add a new fence definition
% \param #1 : fence ID
% \param #2 : fence length
% \param #3 : fence width
% \param #4 : fence color
\newcommand{\addFence}[4]{%
  \addItem{fence@#1}{#2}{#3}{#4}
}%

% \brief Draw a fence given a specific ID definition
% \param #1 : fence ID
% \param #2 : fence x-axis position
% \param #3 : fence y-axis position
% \param #4 : fence orientation [in degrees]
\newcommand{\drawFence}[4]{%

  % Get all parameters
  \drawItem{fence@#1}{#2}{#3}{#4}

  % Draw fence
  \begin{scope}[shift={(\item@x,\item@y)}, rotate=\item@orientation]

    \edef\fence@thickness{0.2}%

    % Draw fence
    \draw[fill=\item@color, draw=black, line width=0.3pt] (\item@xorigin, \item@yorigin) rectangle (\item@xend, \item@yend);%
    \draw[fill=white, draw=black, line width=0.3pt] (\item@xorigin+\fence@thickness, \item@yorigin + \fence@thickness) rectangle (\item@xend - \fence@thickness, (\item@yend - \fence@thickness);%
  \end{scope}%
}

\makeatother
