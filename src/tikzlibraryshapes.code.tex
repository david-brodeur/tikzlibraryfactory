% \brief Draw a triangle
% \param #1 : base
% \param #2 : height
% \param #3 : orientation
% \param #4 : color
\newcommand{\drawTriangle}[4]{%

  \edef\base{#1}%
  \edef\height{#2}%
  \edef\orientation{#3}%
  \edef\color{#4}%

  \begin{scope}[shift={(0,0)}, rotate=\orientation]
    \pgfmathsetmacro{\halfbase}{\base/2}
    \draw[fill=\color] (0,0) -- (-\halfbase, -\height) -- (\halfbase, -\height) -- cycle;
  \end{scope}%
}%
